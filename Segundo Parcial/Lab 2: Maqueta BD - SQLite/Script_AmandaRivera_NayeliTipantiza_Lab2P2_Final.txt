CREATE TABLE Persona(
	ci_persona CHAR(10),
	nombre_persona	 CHAR(30),	
	tel_persona CHAR(10),
	genero_persona CHAR(10),
	fecha_nac_persona DATE NOT NULL,
	correo_persona TEXT	 NOT NULL,
	direccion_persona CHAR(80),
	PRIMARY KEY (ci_persona)
);
CREATE TABLE Hijo(
	ci_hijo CHAR(10),
	nombre_hijo CHAR(30),
	edad_hijo INT NOT NULL,
	ci_persona CHAR(10),
	PRIMARY KEY (ci_hijo),
	FOREIGN KEY (ci_persona) REFERENCES Persona(ci_persona)
);
CREATE TABLE Afiliacion(
	id_afiliacion INT NOT NULL,
	nombre_afiliacion CHAR(20),
	tipo_afiliacion CHAR(20),
	estado_afiliacion CHAR(10),
	ci_persona CHAR(10),
	PRIMARY KEY (id_afiliacion),
	FOREIGN KEY (ci_persona) REFERENCES Persona(ci_persona)
);
CREATE TABLE Base_Legal(
	id_base_legal INT NOT NULL,
	descripcion_base_legal CHAR(50),
	decreto_base_legal CHAR(20),
	estado_base_legal CHAR(10),
	id_afiliacion INT NOT NULL,
	PRIMARY KEY (id_base_legal),
	FOREIGN KEY (id_afiliacion) REFERENCES Afiliacion(id_afiliacion)
);
CREATE TABLE Codigo(
	id_codigo INT NOT NULL,
	nombre_codigo CHAR(30),
	articulo INT NOT NULL,
	estado_codigo CHAR(10),
	detalle_codigo CHAR(30),
	id_base_legal INT NOT NULL,
	PRIMARY KEY (id_codigo),
	FOREIGN KEY (id_base_legal) REFERENCES Base_Legal(id_base_legal)
);
CREATE TABLE Aportacion(
	id_aportacion CHAR(10),
	fecha_aportacion DATE NOT NULL,
	no_aportacion INT NOT NULL,
	detalle_aportacion TEXT NOT NULL,
	monto_aportacion REAL,
	ci_persona CHAR(10),
	PRIMARY KEY (id_aportacion),
	FOREIGN KEY (ci_persona) REFERENCES Persona(ci_persona)
);
CREATE TABLE Empresa(
	id_empresa INT NOT NULL,
	nombre_empresa CHAR(40),
	detalle_empresa CHAR(40),
	direccion_empresa CHAR(50),
	responsable_empresa CHAR(30),
	PRIMARY KEY (id_empresa)
);
CREATE TABLE Usuario(
	id_usuario CHAR(10),
	nombre_usuario CHAR(20),
	contrasenia_usuario CHAR(50),
	ci_persona CHAR(10),
	PRIMARY KEY (id_usuario),
	FOREIGN KEY (ci_persona) REFERENCES Persona(ci_persona)
);
CREATE TABLE Perfil(
	id_perfil INT NOT NULL,
	descripcion_perfil CHAR(50),
	nombre_perfil CHAR(10),
	PRIMARY KEY (id_perfil)
);
CREATE TABLE Rol(
	cod_rol INT NOT NULL,
	nombre_rol CHAR(20),
	detalle_rol CHAR(20),
	PRIMARY KEY (cod_rol)
);
.header on
.mode column
